// extractWidgetInfo extracts WidgetInfo from a widget.
func extractWidgetInfo(w widget.PreferredSizeLocateableWidget) WidgetInfo {
	widgetObj := w.GetWidget()
	info := WidgetInfo{
		Widget:   w,
		Rect:     widgetObj.Rect,
		Visible:  widgetObj.IsVisible(),
		Disabled: widgetObj.Disabled,
		State:    make(map[string]string),
		CustomData: make(map[string]string),
	}

	// Get widget type name from reflection
	info.Type = reflect.TypeOf(w).Elem().Name()

	// Extract custom data
	if widgetObj.CustomData != nil {
		info.CustomData = extractCustomData(widgetObj.CustomData)
	}

	// Extract widget-specific state
	info.State = extractWidgetState(w)

	return info
}

package internal

import (
	"encoding/xml"
	"reflect"
	"strings"

// ExtractCustomData flattens custom data into a string map.
// Supports: structs with xml tags, structs without tags (field names),
// map[string]string, and primitive values.
func ExtractCustomData(data any) map[string]string {
	if data == nil {
		return nil
	}

	result := make(map[string]string)

	switch v := data.(type) {
	case map[string]string:
		// Direct map - copy keys
		for key, val := range v {
			result[key] = val
		}
		return result

	case string:
		result["custom_data"] = v
		return result

	case int, int8, int16, int32, int64:
		result["custom_data"] = formatInt(v)
		return result

	case float32, float64:
		result["custom_data"] = formatFloat(v)
		return result

	case bool:
		result["custom_data"] = formatBool(v)
		return result
	}

	// Handle struct types
	rv := reflect.ValueOf(data)
	if rv.Kind() == reflect.Ptr {
		rv = rv.Elem()
	}
	if rv.Kind() == reflect.Struct {
		return extractStructFields(rv, result, "")
	}

	return result
}

func extractStructFields(rv reflect.Value, result map[string]string, prefix string) map[string]string {
	rt := rv.Type()
	for i := 0; i < rt.NumField(); i++ {
		field := rt.Field(i)
		fv := rv.Field(i)

		if !fv.CanInterface() {
			continue
		}

		// Get attribute name from xml tag or field name
		name := fieldNameFromXMLTag(field, prefix)

		// Handle nested structs
		if fv.Kind() == reflect.Struct {
			extractStructFields(fv, result, name+".")
			continue
		}

		// Handle pointer to struct
		if fv.Kind() == reflect.Ptr && !fv.IsNil() && fv.Elem().Kind() == reflect.Struct {
			extractStructFields(fv.Elem(), result, name+".")
			continue
		}

		// Set value
		result[name] = formatValue(fv.Interface())
	}
	return result
}

func fieldNameFromXMLTag(field reflect.StructField, prefix string) string {
	tag := field.Tag.Get("xml")
	if tag == "" {
		// No xml tag - use original field name
		return prefix + field.Name
	}

	// Parse xml tag: "name,attr" or just "name"
	parts := strings.Split(tag, ",")
	name := parts[0]
	if name == "" || name == "-" {
		return prefix + field.Name
	}
	return prefix + name
}

func formatValue(v any) string {
	switch val := v.(type) {
	case string:
		return val
	case int, int8, int16, int32, int64:
		return formatInt(val)
	case uint, uint8, uint16, uint32, uint64:
		return formatUint(val)
	case float32, float64:
		return formatFloat(val)
	case bool:
		return formatBool(val)
	default:
		return ""
	}
}

func formatInt(v any) string {
	return strings.TrimSpace(strings.ToLower(
		strings.ReplaceAll(strings.ReplaceAll(
			strings.ReplaceAll(reflect.ValueOf(v).String(),
				"int", ""), "8", ""), "16", "")))
}

package internal

import (
	"github.com/ebitenui/ebitenui/widget"
)

// ExtractWidgetState extracts type-specific state from widgets.
func ExtractWidgetState(w widget.PreferredSizeLocateableWidget) map[string]string {
	state := make(map[string]string)

	switch v := w.(type) {
	case *widget.Button:
		state["text"] = v.Text()
		state["state"] = widgetStateToString(v.State())
		state["toggle"] = formatBool(v.ToggleMode)

	case *widget.TextInput:
		state["text"] = v.Text()
		state["cursor"] = formatInt(v.CursorPosition())
		state["focused"] = formatBool(v.Focused())

	case *widget.Checkbox:
		state["checked"] = widgetStateToString(v.State())
		state["triState"] = formatBool(v.TriState())

	case *widget.Slider:
		state["value"] = formatInt(v.Current)
		state["min"] = formatInt(v.Min)
		state["max"] = formatInt(v.Max)

	case *widget.Label:
		state["text"] = v.Label

	case *widget.ProgressBar:
		state["value"] = formatInt(v.Value)
		state["min"] = formatInt(v.Min)
		state["max"] = formatInt(v.Max)
	}

	return state
}

func widgetStateToString(s widget.WidgetState) string {
	switch s {
	case widget.WidgetUnchecked:
		return "unchecked"
	case widget.WidgetChecked:
		return "checked"
	case widget.WidgetGreyed:
		return "greyed"
	}
	return "unknown"
}

// walkTree traverses the widget tree starting from root,
// returning a flat list of all widgets (depth-first).
func walkTree(root widget.PreferredSizeLocateableWidget) []WidgetInfo {
	var result []WidgetInfo
	walkTreeRecursive(root, &result)
	return result
}

func walkTreeRecursive(w widget.PreferredSizeLocateableWidget, result *[]WidgetInfo) {
	info := extractWidgetInfo(w)
	*result = append(*result, info)

	// Check if widget has children (Container)
	if container, ok := w.(*widget.Container); ok {
		for _, child := range container.Children() {
			walkTreeRecursive(child, result)
		}
	}
}

package autoui

import (
	"encoding/xml"
	"image"
	"io"
	"strings"
)

// WidgetNode represents a widget for XML marshaling.
// The element name is set dynamically based on widget type.
type WidgetNode struct {
	XMLName  xml.Name // Dynamic: "Button", "Container", etc.
	Attrs    []xml.Attr
	Children []*WidgetNode
}

// marshalWidgetXML converts a single WidgetInfo to XML bytes.
func marshalWidgetXML(info WidgetInfo) ([]byte, error) {
	node := widgetInfoToNode(info)
	return xml.Marshal(node)
}

// marshalWidgetTreeXML converts a flat widget list to tree XML.
// Reconstructs parent-child relationships from list order.
func marshalWidgetTreeXML(widgets []WidgetInfo) ([]byte, error) {
	if len(widgets) == 0 {
		return []byte("<?xml version=\"1.0\" encoding=\"UTF-8\"?><UI></UI>"), nil
	}

	// Build tree structure
	root := widgetInfoToNode(widgets[0])
	buildNodeTree(root, widgets[1:])

	// Wrap in UI element
	uiNode := &WidgetNode{
		XMLName: xml.Name{Local: "UI"},
		Children: []*WidgetNode{root},
	}

	return xml.MarshalIndent(uiNode, "", "  ")
}

func widgetInfoToNode(info WidgetInfo) *WidgetNode {
	node := &WidgetNode{
		XMLName: xml.Name{Local: info.Type},
		Attrs:   make([]xml.Attr, 0, 10),
	}

	// Add position attributes
	node.Attrs = append(node.Attrs,
		xml.Attr{Name: xml.Name{Local: "x"}, Value: formatInt(info.Rect.Min.X)},
		xml.Attr{Name: xml.Name{Local: "y"}, Value: formatInt(info.Rect.Min.Y)},
		xml.Attr{Name: xml.Name{Local: "width"}, Value: formatInt(info.Rect.Dx())},
		xml.Attr{Name: xml.Name{Local: "height"}, Value: formatInt(info.Rect.Dy())},
		xml.Attr{Name: xml.Name{Local: "visible"}, Value: formatBool(info.Visible)},
		xml.Attr{Name: xml.Name{Local: "disabled"}, Value: formatBool(info.Disabled)},
	)

	// Add state attributes (sorted for consistency)
	for key, val := range sortedMap(info.State) {
		node.Attrs = append(node.Attrs, xml.Attr{Name: xml.Name{Local: key}, Value: val})
	}

	// Add custom data attributes (sorted for consistency)
	for key, val := range sortedMap(info.CustomData) {
		node.Attrs = append(node.Attrs, xml.Attr{Name: xml.Name{Local: key}, Value: val})
	}

	return node
}

// MarshalXML implements custom XML marshaling for dynamic element name.
func (n *WidgetNode) MarshalXML(e *xml.Encoder, start xml.StartElement) error {
	start.Name = n.XMLName
	start.Attr = n.Attrs

	if err := e.EncodeToken(start); err != nil {
		return err
	}

	for _, child := range n.Children {
		if err := e.Encode(child); err != nil {
			return err
		}
	}

	return e.EncodeToken(start.End())
}

// buildNodeTree reconstructs tree from flat list.
// Uses nesting level tracking based on container children.
func buildNodeTree(root *WidgetNode, remaining []WidgetInfo) {
	// Implementation depends on container depth tracking
	// For simple cases, all non-containers become children of previous container
	for _, info := range remaining {
		child := widgetInfoToNode(info)
		if info.Type == "Container" {
			// Containers become children of root, subsequent widgets go to this container
			root.Children = append(root.Children, child)
			root = child // Switch context to new container
		} else {
			root.Children = append(root.Children, child)
		}
	}
}

func sortedMap(m map[string]string) []stringKeyValue {
	keys := make([]string, 0, len(m))
	for k := range m {
		keys = append(keys, k)
	}
	sort.Strings(keys)

	result := make([]stringKeyValue, len(keys))
	for i, k := range keys {
		result[i] = stringKeyValue{Key: k, Value: m[k]}
	}
	return result
}

type stringKeyValue struct {
	Key   string
	Value string
}

package autoui

import (
	"encoding/json"
	"fmt"
	"image"
	"strings"
)

// findByQuery matches widgets using simple key=value syntax.
func findByQuery(widgets []WidgetInfo, query string) []WidgetInfo {
	// Parse key=value
	parts := strings.SplitN(query, "=", 2)
	if len(parts) != 2 {
		return nil
	}
	key := parts[0]
	value := parts[1]

	return filterWidgets(widgets, map[string]string{key: value})
}

// findByQueryJSON matches widgets using JSON criteria (AND logic).
func findByQueryJSON(widgets []WidgetInfo, query string) []WidgetInfo {
	var criteria map[string]string
	if err := json.Unmarshal([]byte(query), &criteria); err != nil {
		return nil
	}
	return filterWidgets(widgets, criteria)
}

// filterWidgets returns widgets matching all criteria.
func filterWidgets(widgets []WidgetInfo, criteria map[string]string) []WidgetInfo {
	var results []WidgetInfo

	for _, w := range widgets {
		if matchesCriteria(w, criteria) {
			results = append(results, w)
		}
	}

	return results
}

// matchesCriteria checks if widget matches all key=value pairs.
func matchesCriteria(w WidgetInfo, criteria map[string]string) bool {
	for key, expectedValue := range criteria {
		actualValue := getWidgetAttributeValue(w, key)
		if actualValue != expectedValue {
			return false
		}
	}
	return true
}

// getWidgetAttributeValue retrieves attribute value by key name.
func getWidgetAttributeValue(w WidgetInfo, key string) string {
	switch key {
	case "type":
		return w.Type
	case "x":
		return formatInt(w.Rect.Min.X)
	case "y":
		return formatInt(w.Rect.Min.Y)
	case "width":
		return formatInt(w.Rect.Dx())
	case "height":
		return formatInt(w.Rect.Dy())
	case "visible":
		return formatBool(w.Visible)
	case "disabled":
		return formatBool(w.Disabled)
	default:
		// Check state then custom data
		if val, ok := w.State[key]; ok {
			return val
		}
		if val, ok := w.CustomData[key]; ok {
			return val
		}
		return ""
	}
}

// findAt returns the widget containing the given coordinates.
// Returns nil if no widget found.
func findAt(widgets []WidgetInfo, x, y int) *WidgetInfo {
	// Search in reverse order (top-most widgets first)
	for i := len(widgets) - 1; i >= 0; i-- {
		w := widgets[i]
		pt := image.Point{X: x, Y: y}
		if pt.In(w.Rect) {
			return &w
		}
	}
	return nil
}

package autoui

import (
	"github.com/antchfx/xmlquery"
)

// queryXPath executes an XPath 1.0 expression on the widget tree.
// Returns matching widgets as WidgetInfo slice.
func queryXPath(widgets []WidgetInfo, xpathExpr string) ([]WidgetInfo, error) {
	// Generate XML from widgets
	xmlBytes, err := marshalWidgetTreeXML(widgets)
	if err != nil {
		return nil, fmt.Errorf("failed to marshal widget tree: %w", err)
	}

	// Parse XML document
	doc, err := xmlquery.Parse(strings.NewReader(string(xmlBytes)))
	if err != nil {
		return nil, fmt.Errorf("failed to parse XML: %w", err)
	}

	// Execute XPath query
	nodes, err := xmlquery.QueryAll(doc, xpathExpr)
	if err != nil {
		return nil, fmt.Errorf("XPath query failed: %w", err)
	}

	// Convert matched XML nodes back to WidgetInfo
	return xmlNodesToWidgetInfo(nodes, widgets), nil
}

// xmlNodesToWidgetInfo maps XPath result nodes back to original widgets.
// Uses type + position matching for identification.
func xmlNodesToWidgetInfo(nodes []*xmlquery.Node, widgets []WidgetInfo) []WidgetInfo {
	var results []WidgetInfo

	for _, node := range nodes {
		// Find matching widget by type and position attributes
		typeName := node.Data
		x := node.SelectAttr("x")
		y := node.SelectAttr("y")

		for _, w := range widgets {
			if w.Type == typeName &&
				formatInt(w.Rect.Min.X) == x &&
				formatInt(w.Rect.Min.Y) == y {
				results = append(results, w)
				break
			}
		}
	}

	return results
}

package autoui

import (
	"fmt"
	"reflect"
)

// Supported method signatures for whitelist.
var supportedSignatures = []reflect.Type{
	reflect.TypeOf(func() {}),
	reflect.TypeOf(func(bool) {}),
	reflect.TypeOf(func(int) {}),
	reflect.TypeOf(func(float64) {}),
	reflect.TypeOf(func(string) {}),
}

// invokeMethod calls a method on a widget via reflection.
// Only whitelisted signatures are allowed for safety.
func invokeMethod(w widget.PreferredSizeLocateableWidget, methodName string, args []any) error {
	rv := reflect.ValueOf(w)
	method := rv.MethodByName(methodName)

	if !method.IsValid() {
		return fmt.Errorf("method %s not found on %s", methodName, reflect.TypeOf(w).Elem().Name())
	}

	// Check whitelist
	methodType := method.Type()
	if !isWhitelistedSignature(methodType) {
		return fmt.Errorf("method %s has unsupported signature", methodName)
	}

	// Convert and validate arguments
	convertedArgs, err := convertArgs(args, methodType)
	if err != nil {
		return fmt.Errorf("argument conversion failed: %w", err)
	}

	// Invoke method
	result := method.Call(convertedArgs)

	// Check for panics (already recovered by reflection)
	if len(result) > 0 && result[0].Kind() == reflect.Interface {
		if errVal := result[0].Interface(); errVal != nil {
			if err, ok := errVal.(error); ok {
				return err
			}
		}
	}

	return nil
}

func isWhitelistedSignature(t reflect.Type) bool {
	for _, sig := range supportedSignatures {
		if t.AssignableTo(sig) {
			return true
		}
	}
	return false
}

func convertArgs(args []any, methodType reflect.Type) ([]reflect.Value, error) {
	if len(args) != methodType.NumIn() {
		return nil, fmt.Errorf("expected %d args, got %d", methodType.NumIn(), len(args))
	}

	converted := make([]reflect.Value, len(args))
	for i, arg := range args {
		targetType := methodType.In(i)
		converted[i], err = convertArg(arg, targetType)
		if err != nil {
			return nil, err
		}
	}

	return converted, nil
}

func convertArg(arg any, targetType reflect.Type) (reflect.Value, error) {
	argType := reflect.TypeOf(arg)

	// Direct match
	if argType.AssignableTo(targetType) {
		return reflect.ValueOf(arg), nil
	}

	// Numeric conversions
	switch targetType.Kind() {
	case reflect.Bool:
		if b, ok := arg.(bool); ok {
			return reflect.ValueOf(b), nil
		}

	case reflect.Int, reflect.Int8, reflect.Int16, reflect.Int32, reflect.Int64:
		switch v := arg.(type) {
		case int:
			return reflect.ValueOf(v).Convert(targetType), nil
		case float64:
			return reflect.ValueOf(int(v)).Convert(targetType), nil
		}

	case reflect.Float32, reflect.Float64:
		switch v := arg.(type) {
		case float64:
			return reflect.ValueOf(v).Convert(targetType), nil
		case int:
			return reflect.ValueOf(float64(v)).Convert(targetType), nil
		}

	case reflect.String:
		if s, ok := arg.(string); ok {
			return reflect.ValueOf(s), nil
		}
	}

	return reflect.Value{}, fmt.Errorf("cannot convert %T to %v", arg, targetType)
}

import (
	"image"
	"image/color"
	"sync"
	"time"

	"github.com/hajimehoshi/ebiten/v2"
	"github.com/hajimehoshi/ebiten/v2/vector"

// Highlight represents a visual highlight rectangle.
type Highlight struct {
	Rect      image.Rectangle
	Color     color.Color
	ExpiresAt time.Time
}

// highlightManager manages active highlights.
type highlightManager struct {
	mu        sync.RWMutex
	highlights []Highlight
	duration  time.Duration
	color     color.Color
}

var globalHighlightManager = &highlightManager{
	duration: 3 * time.Second,
	color:    color.RGBA{255, 0, 0, 255}, // Red
}

func newHighlightManager() *highlightManager {
	return &highlightManager{
		duration: 3 * time.Second,
		color:    color.RGBA{255, 0, 0, 255},
	}
}

func (m *highlightManager) add(rect image.Rectangle) {
	m.mu.Lock()
	defer m.mu.Unlock()

	m.highlights = append(m.highlights, Highlight{
		Rect:      rect,
		Color:     m.color,
		ExpiresAt: time.Now().Add(m.duration),
	})
}

func (m *highlightManager) clear() {
	m.mu.Lock()
	defer m.mu.Unlock()
	m.highlights = nil
}

func (m *highlightManager) getActive() []Highlight {
	m.mu.Lock()
	defer m.mu.Unlock()

	// Filter expired
	now := time.Now()
	active := make([]Highlight, 0, len(m.highlights))
	for _, h := range m.highlights {
		if now.Before(h.ExpiresAt) {
			active = append(active, h)
		}
	}
	m.highlights = active
	return active
}

// DrawHighlights renders all active highlights onto the screen.
// Call this in your game's Draw method.
func DrawHighlights(screen *ebiten.Image) {
	highlights := globalHighlightManager.getActive()
	for _, h := range highlights {
		vector.StrokeRect(
			screen,
			float32(h.Rect.Min.X),
			float32(h.Rect.Min.Y),
			float32(h.Rect.Dx()),
			float32(h.Rect.Dy()),
			2, // Stroke width
			h.Color,
			false,
		)
	}
}

// SetHighlightDuration configures how long highlights persist.
func SetHighlightDuration(d time.Duration) {
	globalHighlightManager.mu.Lock()
	defer globalHighlightManager.mu.Unlock()
	globalHighlightManager.duration = d
}

		return

	Method string   `json:"method"` // Method name to invoke
	Args   []any    `json:"args"`   // Arguments for the method
}

func handleCallCommand(ctx autoebiten.CommandContext) {
	if uiReference == nil {
		ctx.Respond("error: autoui not registered")
		return
	}

	request := ctx.Request()
	if request == "" {
		ctx.Respond("error: empty request")
		return
	}

	var callReq CallRequest
	if err := json.Unmarshal([]byte(request), &callReq); err != nil {
		ctx.Respond("error: " + err.Error())
		return
	}

	// Find target widget
	widgets := walkTree(uiReference.Container)
	var results []WidgetInfo
	if strings.HasPrefix(callReq.Target, "{") {
		results = findByQueryJSON(widgets, callReq.Target)
	} else {
		results = findByQuery(widgets, callReq.Target)
	}

	if len(results) == 0 {
		ctx.Respond("error: target widget not found")
		return
	}

	// Invoke on first match
	target := results[0]
	err := invokeMethod(target.Widget, callReq.Method, callReq.Args)
	if err != nil {
		ctx.Respond("error: " + err.Error())
		return
	}

	ctx.Respond(fmt.Sprintf("success: invoked %s on %s", callReq.Method, target.Type))
}

package autoui

import (
	"strings"

	"github.com/s3cy/autoebiten"
)

func handleHighlightCommand(ctx autoebiten.CommandContext) {
	if uiReference == nil {
		ctx.Respond("error: autoui not registered")
		return
	}

	request := ctx.Request()

	// Handle clear
	if request == "clear" {
		globalHighlightManager.clear()
		ctx.Respond("success: highlights cleared")
		return
	}

	widgets := walkTree(uiReference.Container)

	// Handle coordinates "x,y"
	if strings.Contains(request, ",") && !strings.Contains(request, "=") {
		x, y := parseCoordinates(request)
		widget := findAt(widgets, x, y)
		if widget == nil {
			ctx.Respond("error: no widget at coordinates")
			return
		}
		globalHighlightManager.add(widget.Rect)
		ctx.Respond("success: highlighting widget")
		return
	}

	// Handle query (same as find)
	var results []WidgetInfo
	if strings.HasPrefix(request, "{") {
		results = findByQueryJSON(widgets, request)
	} else if request != "" {
		results = findByQuery(widgets, request)
	}

	if len(results) == 0 {
		ctx.Respond("error: no matching widgets")
		return
	}

	// Add highlights for all matches
	for _, w := range results {
		globalHighlightManager.add(w.Rect)
	}

	ctx.Respond(fmt.Sprintf("success: highlighting %d widgets", len(results)))
}

func parseCoordinates(s string) (int, int) {
	parts := strings.Split(s, ",")
	if len(parts) != 2 {
		return 0, 0
	}
	return parseIntOrZero(parts[0]), parseIntOrZero(parts[1])
}
